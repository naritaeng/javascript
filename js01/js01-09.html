        let a = 10
        let b = 7
        let c = 11
        let d = 13
        let e = 9
        console.log(`a=${a},b=${b},c=${c},d=${d},e=${e} 일 때`)
        console.log(`a+=1 => ${a+=1}`) // 12
        console.log(`a+=1 => ${a+=1}`) // 14
        console.log(`b-=2 => ${b-=2}`) // 5
        console.log(`b-=2 => ${b-=2}`) // 3
        console.log(`c*=2 => ${c*=2}`) // 22
        console.log(`c*=2 => ${c*=2}`) // 44
        console.log(`d/=2 => ${d/=2}`) // 6.5
        console.log(`d/=2 => ${d/=2}`) // 3.25
        console.log(`e%=2 => ${e%=2}`) // 1
        console.log(`e%=2 => ${e%=2}`) // 1

        const 변하지 않는 값을 대입
        선언할 때 : const 이름 = 값;
        let 은 변할 가능성이 있는 값을 대입
        선언할 때 : let 이름 = 값;
