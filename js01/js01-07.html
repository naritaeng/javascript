        console.log('typeof("") => '+typeof("")) // string
        console.log('typeof("문자열") => '+typeof("문자열")) // string
        console.log('typeof("1234") => '+typeof("1234")) // string
        console.log('typeof(1234) => '+typeof(1234)) //number
        console.log('typeof(true) => '+typeof(true)) // boolean
        console.log('typeof("true") => '+typeof("true")) // string
        console.log('typeof(false) => '+typeof(false)) // boolean
        console.log('typeof(null) => '+typeof(null)) // object
        console.log('typeof(undefined) => '+typeof(undefined)) // undefined
